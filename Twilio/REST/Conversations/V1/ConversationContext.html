def initialize(version, sid)

def fetch
  params = {}
  
  payload = @version.fetch(
      'GET',
      @uri,
      params,
  )
  
  return ConversationInstance.new(
      @version,
      payload,
      sid: @solution['sid'],
  )
end

def participants(sid=:unset)
  if sid != :unset
    return ParticipantContext.new(
        @version,
        @solution[:sid],
        sid,
    )
  end
  
  unless @participants
    @participants = ParticipantList.new(
        @version,
        conversation_sid: @solution[:sid],
    )
  end
  
  @participants
end
