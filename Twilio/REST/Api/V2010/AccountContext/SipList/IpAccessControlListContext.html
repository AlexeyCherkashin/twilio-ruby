end

def ip_addresses(sid=:unset)
  if sid != :unset
    return IpAddressContext.new(
        @version,
        @solution[:account_sid],
        @solution[:sid],
        sid,
    )
  end
  
  unless @ip_addresses
    @ip_addresses = IpAddressList.new(
        @version,
        account_sid: @solution[:account_sid],
        ip_access_control_list_sid: @solution[:sid],
    )
  end
  
  @ip_addresses
end
