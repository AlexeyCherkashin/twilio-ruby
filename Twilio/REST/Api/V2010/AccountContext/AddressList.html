def create(customer_name: nil, street: nil, city: nil, region: nil, postal_code: nil, iso_country: nil, friendly_name: nil)
  data = {
      'CustomerName' => customer_name,
      'Street' => street,
      'City' => city,
      'Region' => region,
      'PostalCode' => postal_code,
      'IsoCountry' => iso_country,
      'FriendlyName' => friendly_name,
  }
  
  payload = @version.create(
      'POST',
      @uri,
      data: data
  )
  
  return AddressInstance.new(
      @version,
      payload,
      account_sid: @solution['account_sid'],
  )
end

- (Object) each

When passed a block, yields AddressInstance records from the API. This operation lazily loads records as efficiently as possible until the limit is reached.

Parameters:

  • customer_name (String) —
    The customer_name
  • friendly_name (String) —
    The friendly_name
  • iso_country (String) —
    The iso_country
  • limit (Integer) —
    Upper limit for the number of records to return. stream() guarantees to never return more than limit. Default is no limit
  • page_size (Integer) —
    Number of records to fetch per request, when not set will use the default value of 50 records. If no page_size is defined but a limit is defined, stream() will attempt to read the limit with the most efficient page size, i.e. min(limit, 1000)

def each
  limits = @version.read_limits
  
  page = self.page(
      page_size: limits['page_size'],
  )
  
  @version.stream(page,
                  limit: limits['limit'],
                  page_limit: limits['page_limit']).each {|x| yield x}
end

def page(customer_name: nil, friendly_name: nil, iso_country: nil, page_token: nil, page_number: nil, page_size: nil)
  params = {
      'CustomerName' => customer_name,
      'FriendlyName' => friendly_name,
      'IsoCountry' => iso_country,
      'PageToken' => page_token,
      'Page' => page_number,
      'PageSize' => page_size,
  }
  response = @version.page(
      'GET',
      @uri,
      params
  )
  return AddressPage.new(@version, response, @solution)
end

def stream(customer_name: nil, friendly_name: nil, iso_country: nil, limit: nil, page_size: nil)
  limits = @version.read_limits(limit, page_size)
  
  page = self.page(
      customer_name: customer_name,
      friendly_name: friendly_name,
      iso_country: iso_country,
      page_size: limits['page_size'],
  )
  
  @version.stream(page, limit: limits['limit'], page_limit: limits['page_limit'])
end
