def create(attributes: nil, workflow_sid: nil, timeout: nil, priority: nil)
  data = {
      'Attributes' => attributes,
      'WorkflowSid' => workflow_sid,
      'Timeout' => timeout,
      'Priority' => priority,
  }
  
  payload = @version.create(
      'POST',
      @uri,
      data: data
  )
  
  return TaskInstance.new(
      @version,
      payload,
      workspace_sid: @solution['workspace_sid'],
  )
end

When passed a block, yields TaskInstance records from the API. This operation lazily loads records as efficiently as possible until the limit is reached.

Parameters:

  • priority (String) —
    The priority
  • assignment_status (task.Status) —
    The assignment_status
  • workflow_sid (String) —
    The workflow_sid
  • workflow_name (String) —
    The workflow_name
  • task_queue_sid (String) —
    The task_queue_sid
  • task_queue_name (String) —
    The task_queue_name
  • limit (Integer) —
    Upper limit for the number of records to return. stream() guarantees to never return more than limit. Default is no limit
  • page_size (Integer) —
    Number of records to fetch per request, when not set will use the default value of 50 records. If no page_size is defined but a limit is defined, stream() will attempt to read the limit with the most efficient page size, i.e. min(limit, 1000)

def each
  limits = @version.read_limits
  
  page = self.page(
      page_size: limits['page_size'],
  )
  
  @version.stream(page,
                  limit: limits['limit'],
                  page_limit: limits['page_limit']).each {|x| yield x}
end

def page(priority: nil, assignment_status: nil, workflow_sid: nil, workflow_name: nil, task_queue_sid: nil, task_queue_name: nil, page_token: nil, page_number: nil, page_size: nil)
  params = {
      'Priority' => priority,
      'AssignmentStatus' => assignment_status,
      'WorkflowSid' => workflow_sid,
      'WorkflowName' => workflow_name,
      'TaskQueueSid' => task_queue_sid,
      'TaskQueueName' => task_queue_name,
      'PageToken' => page_token,
      'Page' => page_number,
      'PageSize' => page_size,
  }
  response = @version.page(
      'GET',
      @uri,
      params
  )
  return TaskPage.new(@version, response, @solution)
end

def stream(priority: nil, assignment_status: nil, workflow_sid: nil, workflow_name: nil, task_queue_sid: nil, task_queue_name: nil, limit: nil, page_size: nil)
  limits = @version.read_limits(limit, page_size)
  
  page = self.page(
      priority: priority,
      assignment_status: assignment_status,
      workflow_sid: workflow_sid,
      workflow_name: workflow_name,
      task_queue_sid: task_queue_sid,
      task_queue_name: task_queue_name,
      page_size: limits['page_size'],
  )
  
  @version.stream(page, limit: limits['limit'], page_limit: limits['page_limit'])
end
